MetronicApp.controller("helloCtrl2", ["$scope", "$interval", function($scope, $interval) {
	$scope.arr = [];

	var cnt = 1;
	$scope.start = function() {
		$interval(function() {
			if (cnt % 10 == 0) {
				$scope.arr = [];
			}
			$scope.arr.push({
				index: cnt++,
				dates: new Date
			});
		}, 1000);
	}

	$scope.showIndex = function(index) {
		alert("line number:" + index);
	}
}]);
